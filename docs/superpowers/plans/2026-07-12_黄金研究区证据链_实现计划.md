```python
def segment_rolling_corr(rolling, expected, min_len):
    """按 corr_tone 三态把滚动相关序列切成连续段;短段并入邻段防碎片。"""
    points = [row for row in rolling if row.get("corr") is not None]
    if not points:
        return []
    runs = []
    for row in points:
        state = corr_tone(row["corr"], expected=expected)
        if runs and runs[-1]["state"] == state:
            runs[-1]["rows"].append(row)
        else:
            runs.append({"state": state, "rows": [row]})
    merged = []
    for run in runs:
        if merged and (len(run["rows"]) < min_len or merged[-1]["state"] == run["state"]):
            merged[-1]["rows"].extend(run["rows"])
            continue
        merged.append(run)
    if len(merged) > 1 and len(merged[0]["rows"]) < min_len:
        merged[1]["rows"][0:0] = merged[0]["rows"]
        merged.pop(0)
    segments = []
    for run in merged:
        rows = run["rows"]
        values = [row["corr"] for row in rows]
        segments.append({
            "start": rows[0]["date"],
            "end": rows[-1]["date"],
            "_start": rows[0]["_date"],
            "_end": rows[-1]["_date"],
            "state": run["state"],
            "corr": sum(values) / len(values),
            "n": len(rows),
        })
    return segments
```

```python
def make_corr_chart(series_defs, aria_label, width=720, height=180, limit=96,
                    segments=None, chart_id=None):
    top_pad = 18
    bottom_pad = 34
    left_pad = 38
    right_pad = 16
    ribbon_height = 46 if segments else 0
    svg_height = height + ribbon_height
    chart_width = width - left_pad - right_pad
    chart_height = height - top_pad - bottom_pad

    def y_for(value):
        clipped = max(-1, min(1, value))
        return top_pad + chart_height * (1 - (clipped + 1) / 2)

    zero_y = y_for(0)
    primary_points = [
        row for row in series_defs[0]["rows"] if row.get("corr") is not None
    ][-limit:]

    def x_at(index):
        return left_pad + chart_width * index / max(1, len(primary_points) - 1)

    paths = []
    legends = []
    visible_points = []
    for series_index, series in enumerate(series_defs):
        points = [row for row in series["rows"] if row.get("corr") is not None][-limit:]
        if len(points) < 2:
            continue
        visible_points.extend(points)
        coords = []
        for i, row in enumerate(points):
            x = left_pad + chart_width * i / (len(points) - 1)
            coords.append((x, y_for(row["corr"])))
        path = " ".join(f"{'M' if i == 0 else 'L'} {x:.1f} {y:.1f}" for i, (x, y) in enumerate(coords))
        dash = ' stroke-dasharray="5 4"' if series.get("dash") else ""
        paths.append(
            f'<path d="{path}" fill="none" stroke="{series["color"]}" stroke-width="2.4"{dash}></path>'
            f'<circle cx="{coords[-1][0]:.1f}" cy="{coords[-1][1]:.1f}" r="3.2" fill="{series["color"]}"></circle>'
        )
        value_html = fmt_corr(points[-1]["corr"])
        if chart_id and series_index == 0:
            value_html = f'<b data-hover-slot="corr">{value_html}</b>'
        legends.append(
            f'<span><i style="background:{series["color"]}"></i>{escape(series["label"])} {value_html}</span>'
        )

    if not paths:
        return '<div class="empty-chart">暂无足够滚动相关数据</div>'

    band_rects = []
    ribbon_cells = []
    if segments and len(primary_points) >= 2:
        ribbon_y = height - bottom_pad + 24
        for seg in segments:
            start_idx = next(
                (i for i, row in enumerate(primary_points) if row["_date"] >= seg["_start"]), None)
            end_idx = next(
                (i for i in range(len(primary_points) - 1, -1, -1)
                 if primary_points[i]["_date"] <= seg["_end"]), None)
            if start_idx is None or end_idx is None or end_idx < start_idx:
                continue
            x1 = x_at(start_idx)
            x2 = x_at(end_idx)
            seg_width = max(x2 - x1, 1.0)
            fill, opacity = SEGMENT_BAND_FILLS[seg["state"]]
            band_rects.append(
                f'<rect class="regime-band" x="{x1:.1f}" y="{top_pad}" width="{seg_width:.1f}" '
                f'height="{chart_height}" fill="{fill}" fill-opacity="{opacity}"></rect>'
            )
            ribbon_fill, ribbon_opacity = SEGMENT_RIBBON_FILLS[seg["state"]]
            gold_part = (
                f" · 金价 {fmt_pct(seg['gold_return'])}"
                if seg.get("gold_return") is not None else ""
            )
            title = (
                f"{seg['start']} 至 {seg['end']} · {seg['state']} · "
                f"相关 {fmt_corr(seg['corr'])}{gold_part} · n={seg['n']}"
            )
            cell_text = (
                f'<text x="{(x1 + x2) / 2:.1f}" y="{ribbon_y + 12:.1f}" text-anchor="middle" '
                f'class="ribbon-label">{fmt_corr(seg["corr"])}'
                f'<tspan class="ribbon-n"> · n{seg["n"]}</tspan></text>'
                if seg_width >= 40 else ""
            )
            ribbon_cells.append(
                f'<rect class="regime-ribbon-cell" x="{x1:.1f}" y="{ribbon_y:.1f}" '
                f'width="{seg_width:.1f}" height="16" fill="{ribbon_fill}" '
                f'fill-opacity="{ribbon_opacity}"><title>{escape(title)}</title></rect>'
                f"{cell_text}"
            )

    visible_points.sort(key=lambda row: row["_date"])
    first_label = visible_points[0]["date"][2:7]
    last_label = visible_points[-1]["date"][2:7]

    hover_attr = ""
    hover_json = ""
    hover_line = ""
    hover_date = ""
    if chart_id and len(primary_points) >= 2:
        hover_attr = ' data-hover-chart=""'
        payload = {
            "series": [{"label": series_defs[0]["label"], "key": "corr"}],
            "points": [
                {"date": row["date"], "x": round(x_at(i), 2), "corr": round(row["corr"], 4)}
                for i, row in enumerate(primary_points)
            ],
        }
        payload_json = json.dumps(payload, ensure_ascii=False).replace("<", "\\u003c")
        hover_json = f'<script type="application/json" id="{chart_id}-data">{payload_json}</script>'
        hover_line = (
            f'<line data-hover-line hidden x1="0" x2="0" y1="{top_pad}" '
            f'y2="{height - bottom_pad}" class="hover-line"></line>'
        )
        hover_date = f'<span class="hover-date" data-hover-date>{escape(primary_points[-1]["date"])}</span>'

    return f"""
    <div class="corr-wrap"{hover_attr}>
      <div class="corr-legend">{''.join(legends)}{hover_date}</div>
      <svg class="corr-chart" viewBox="0 0 {width} {svg_height}" role="img" aria-label="{escape(aria_label)}">
        {''.join(band_rects)}
        <line x1="{left_pad}" y1="{y_for(0.5):.1f}" x2="{width - right_pad}" y2="{y_for(0.5):.1f}" class="grid-line"></line>
        <line x1="{left_pad}" y1="{y_for(-0.5):.1f}" x2="{width - right_pad}" y2="{y_for(-0.5):.1f}" class="grid-line"></line>
        <line x1="{left_pad}" y1="{zero_y:.1f}" x2="{width - right_pad}" y2="{zero_y:.1f}" class="zero-axis"></line>
        <line x1="{left_pad}" y1="{top_pad}" x2="{left_pad}" y2="{height - bottom_pad}" class="y-axis"></line>
        <line x1="{left_pad}" y1="{height - bottom_pad}" x2="{width - right_pad}" y2="{height - bottom_pad}" class="x-axis"></line>
        <text x="8" y="{top_pad + 4}" class="chart-label">+1</text>
        <text x="8" y="{y_for(0.5) + 4:.1f}" class="chart-label">+0.5</text>
        <text x="8" y="{zero_y + 4:.1f}" class="chart-label">0</text>
        <text x="8" y="{y_for(-0.5) + 4:.1f}" class="chart-label">-0.5</text>
        <text x="8" y="{height - bottom_pad}" class="chart-label">-1</text>
        <text x="{left_pad}" y="{height - 8}" class="chart-label">{first_label}</text>
        <text x="{width - right_pad}" y="{height - 8}" text-anchor="end" class="chart-label">{last_label}</text>
        {''.join(paths)}
        {hover_line}
        {''.join(ribbon_cells)}
      </svg>
      {hover_json}
    </div>
    """
```

def make_dual_axis_chart(
    rows,
    factor_key,
    factor_label,
    gold_key="gold_price",
    gold_label="黄金价格",
    width=720,
    height=220,
    limit=180,
    factor_color="#366b9f",
    gold_color="#b88728",
    invert_factor=False,
    chart_id=None,
):
    points = [
        row for row in rows
        if row.get(factor_key) is not None and row.get(gold_key) is not None
    ][-limit:]
    if len(points) < 2:
        return '<div class="empty-chart">暂无足够双轴走势数据</div>'

    left_pad = 56
    right_pad = 66
    top_pad = 22
    bottom_pad = 32
    chart_width = width - left_pad - right_pad
    chart_height = height - top_pad - bottom_pad

    factor_values = [row[factor_key] for row in points]
    gold_values = [row[gold_key] for row in points]
    factor_min, factor_max = min(factor_values), max(factor_values)
    gold_min, gold_max = min(gold_values), max(gold_values)
    factor_span = factor_max - factor_min or 1
    gold_span = gold_max - gold_min or 1

    def x_for(index):
        return left_pad + chart_width * index / (len(points) - 1)

    def gold_y(value):
        return top_pad + chart_height * (1 - (value - gold_min) / gold_span)

    def factor_y(value):
        ratio = (value - factor_min) / factor_span
        if invert_factor:
            return top_pad + chart_height * ratio
        return top_pad + chart_height * (1 - ratio)

    factor_coords = [(x_for(i), factor_y(row[factor_key])) for i, row in enumerate(points)]
    gold_coords = [(x_for(i), gold_y(row[gold_key])) for i, row in enumerate(points)]
    factor_path = " ".join(
        f"{'M' if i == 0 else 'L'} {x:.1f} {y:.1f}" for i, (x, y) in enumerate(factor_coords))
    gold_path = " ".join(
        f"{'M' if i == 0 else 'L'} {x:.1f} {y:.1f}" for i, (x, y) in enumerate(gold_coords))
    first_label = points[0]["date"][2:7]
    last_label = points[-1]["date"][2:7]

    mid_y = top_pad + chart_height / 2
    grid_ys = [top_pad + chart_height * ratio for ratio in (0.25, 0.5, 0.75)]
    grid_lines = "".join(
        f'<line x1="{left_pad}" y1="{y:.1f}" x2="{width - right_pad}" y2="{y:.1f}" class="grid-line"></line>'
        for y in grid_ys
    )
    factor_axis_top = factor_min if invert_factor else factor_max
    factor_axis_bottom = factor_max if invert_factor else factor_min
    factor_axis_mid = (factor_min + factor_max) / 2
    invert_note = "(已反转)" if invert_factor else ""

    hover_attr = ""
    hover_json = ""
    hover_line = ""
    hover_date = ""
    factor_value_html = fmt_axis_value(points[-1][factor_key])
    gold_value_html = fmt_axis_value(points[-1][gold_key])
    if chart_id:
        hover_attr = ' data-hover-chart=""'
        payload = {
            "series": [
                {"label": factor_label, "key": "factor"},
                {"label": gold_label, "key": "gold"},
            ],
            "points": [
                {
                    "date": row["date"],
                    "x": round(x_for(i), 2),
                    "factor": row[factor_key],
                    "gold": row[gold_key],
                }
                for i, row in enumerate(points)
            ],
        }
        payload_json = json.dumps(payload, ensure_ascii=False).replace("<", "\\u003c")
        hover_json = f'<script type="application/json" id="{chart_id}-data">{payload_json}</script>'
        hover_line = (
            f'<line data-hover-line hidden x1="0" x2="0" y1="{top_pad}" '
            f'y2="{height - bottom_pad}" class="hover-line"></line>'
        )
        hover_date = f'<span class="hover-date" data-hover-date>{escape(points[-1]["date"])}</span>'
        factor_value_html = f'<b data-hover-slot="factor">{factor_value_html}</b>'
        gold_value_html = f'<b data-hover-slot="gold">{gold_value_html}</b>'

    return f"""
    <div class="dual-axis-wrap"{hover_attr}>
      <div class="dual-axis-legend">
        <span><i style="background:{factor_color}"></i>{escape(factor_label)}{invert_note} {factor_value_html}</span>
        <span><i style="background:{gold_color}"></i>{escape(gold_label)} {gold_value_html}</span>
        {hover_date}
      </div>
      <svg class="dual-axis-chart" viewBox="0 0 {width} {height}" role="img" aria-label="{escape(factor_label)}与{escape(gold_label)}双轴走势{invert_note}">
        {grid_lines}
        <line x1="{left_pad}" y1="{top_pad}" x2="{left_pad}" y2="{height - bottom_pad}" class="left-axis"></line>
        <line x1="{width - right_pad}" y1="{top_pad}" x2="{width - right_pad}" y2="{height - bottom_pad}" class="right-axis"></line>
        <line x1="{left_pad}" y1="{height - bottom_pad}" x2="{width - right_pad}" y2="{height - bottom_pad}" class="x-axis"></line>
        <text x="6" y="{top_pad + 4}" class="chart-label">{fmt_axis_value(gold_max)}</text>
        <text x="6" y="{mid_y + 4:.1f}" class="chart-label">{fmt_axis_value((gold_min + gold_max) / 2)}</text>
        <text x="6" y="{height - bottom_pad}" class="chart-label">{fmt_axis_value(gold_min)}</text>
        <text x="{width - 6}" y="{top_pad + 4}" text-anchor="end" class="chart-label">{fmt_axis_value(factor_axis_top)}</text>
        <text x="{width - 6}" y="{mid_y + 4:.1f}" text-anchor="end" class="chart-label">{fmt_axis_value(factor_axis_mid)}</text>
        <text x="{width - 6}" y="{height - bottom_pad}" text-anchor="end" class="chart-label">{fmt_axis_value(factor_axis_bottom)}{invert_note and ' ↕'}</text>
        <path d="{factor_path}" fill="none" stroke="{factor_color}" stroke-width="2.3"></path>
        <path d="{gold_path}" fill="none" stroke="{gold_color}" stroke-width="2.3"></path>
        <circle cx="{factor_coords[-1][0]:.1f}" cy="{factor_coords[-1][1]:.1f}" r="3.1" fill="{factor_color}"></circle>
        <circle cx="{gold_coords[-1][0]:.1f}" cy="{gold_coords[-1][1]:.1f}" r="3.1" fill="{gold_color}"></circle>
        {hover_line}
        <text x="{left_pad}" y="{height - 8}" class="chart-label">{first_label}</text>
        <text x="{width - right_pad}" y="{height - 8}" text-anchor="end" class="chart-label">{last_label}</text>
      </svg>
      {hover_json}
    </div>
    """
```

      <summary class="evidence-summary">

def make_evidence_relationship_block(relationship, unit_id, min_len, chart_limit, corr_label):
    expected = relationship["expected"]
    segments = segment_rolling_corr(relationship["rolling_corr"], expected, min_len)
    for seg in segments:
        seg["gold_return"] = phase_gold_return(
            relationship["trend_rows"], seg["_start"], seg["_end"])
    invert = expected == "negative"
    dual = make_dual_axis_chart(
        relationship["trend_rows"],
        relationship["factor_key"],
        relationship["factor_label"],
        relationship["gold_key"],
        limit=chart_limit,
        invert_factor=invert,
        chart_id=f"{unit_id}-main",
    )
    series = [{"label": corr_label, "rows": relationship["rolling_corr"], "color": "#237a57"}]
    if relationship.get("short_term"):
        series.append({
            "label": relationship["short_term"]["label"],
            "rows": relationship["short_term"]["rolling_corr"],
            "color": "#366b9f",
            "dash": True,
        })
    corr = make_corr_chart(
        series,
        f"{relationship['name']}与黄金收益滚动相关及分段",
        limit=chart_limit,
        segments=segments,
        chart_id=f"{unit_id}-corr",
    )
    return f"""
      <div class="evidence-chart-block">
        <h4>主关系走势</h4>
        {dual}
        <h4>关系演变</h4>
        {corr}
      </div>
    """


def make_evidence_unit(unit_id, title, source_note, relationship, layer,
                       min_len, chart_limit, spark_limit, corr_label, extra_sections=""):
    spark = make_sparkline(
        relationship["rolling_corr"], "corr",
        color="#237a57", width=64, height=16, limit=spark_limit)
    summary = make_evidence_summary(
        title,
        source_note,
        tone_badge(relationship["latest_tone"]),
        spark,
        f"滚动相关 <strong>{fmt_corr(relationship['latest_corr'])}</strong>",
        f'<span class="quality">{quality_badge(layer["data_quality"])}</span>' if layer else "",
    )
    hook = make_evidence_hook(layer) if layer else ""
    block = make_evidence_relationship_block(
        relationship, unit_id, min_len, chart_limit, corr_label)
    return f"""
    <details class="evidence-unit" id="{unit_id}">
      {summary}
      <div class="evidence-body">
        {hook}
        <p class="evidence-read">{escape(relationship['read'])}</p>
        {block}
        {extra_sections}
        <p class="evidence-footnote">分段与徽章按「{escape(corr_label)}」以现有阈值(顺预期 ≥0.35 有效、逆预期 ≥0.1 失效)自动判定;阈值、窗口与已知局限见「方法与数据」。</p>
      </div>
    </details>
    """

      if (!rect.width) return;

      const viewX = (event.clientX - rect.left) / rect.width * viewWidth;
      const first = payload.points[0].x;
      const last = payload.points[payload.points.length - 1].x;
      const ratio = Math.max(0, Math.min(1, (viewX - first) / ((last - first) || 1)));
      show(payload.points[Math.round(ratio * (payload.points.length - 1))], true);
    };

    svg.addEventListener("pointermove", locate);
    svg.addEventListener("pointerdown", locate);
    svg.addEventListener("pointerleave", (event) => {
      if (event.pointerType === "touch" || event.pointerType === "pen") return;
      show(payload.points[payload.points.length - 1], false);
    });
  });
})();
```
